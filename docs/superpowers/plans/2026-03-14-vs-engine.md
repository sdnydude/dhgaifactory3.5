"""Core distribution math for Verbalized Sampling.

Ported from CHATS-lab verbalized_sampling/selection.py (~300 lines).
Contains: Item, DiscreteDist, repair_weight, postprocess_responses.
"""

import math
import random
import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Callable, Dict, List, Literal, Optional, Tuple


def repair_weight(raw: Any) -> Tuple[float, List[str]]:
    """Repair a raw weight value to [0, 1], tracking all operations.

    Handles: percentages ("45%"), negative values, NaN, Inf, string floats,
    unparseable strings, None. Falls back to 0.0 for anything unsalvageable.

    Args:
        raw: Raw weight value (may be str, float, int, None, malformed)

    Returns:
        Tuple of (repaired_weight, list_of_repair_operations)
    """
    repairs: List[str] = []
    value = raw

    # Handle string inputs
    if isinstance(value, str):
        value = value.strip()

        # Handle percentages ("70%" -> 0.70)
        if value.endswith("%"):
            repairs.append("percentage")
            try:
                value = float(value.rstrip("%")) / 100.0
            except ValueError:
                repairs.append("invalid")
                return (0.0, repairs)

        # Try parsing as float
        try:
            value = float(value)
        except ValueError:
            repairs.append("invalid")
            return (0.0, repairs)

    # Handle numeric types
    try:
        value = float(value)
    except (TypeError, ValueError):
        repairs.append("invalid")
        return (0.0, repairs)

    # Check for invalid numeric values
    if math.isnan(value) or math.isinf(value):
        repairs.append("invalid")
        return (0.0, repairs)

    # Clip negatives to zero
    if value < 0:
        repairs.append("negative")
        value = 0.0

    # Clip > 1 to 1
    if value > 1:
        repairs.append("clip>1")
        value = 1.0

    return (value, repairs)
```

@dataclass(frozen=True)
class Item:
    """A single generated response with its probability.

    Attributes:
        text: The response text
        p: Normalized probability in [0, 1]
        meta: Metadata dict (p_raw, repairs, label, quality_score, etc.)
    """

    text: str
    p: float
    meta: Dict[str, Any] = field(default_factory=dict)

    def __post_init__(self):
        object.__setattr__(self, "meta", dict(self.meta))
        if not (0 <= self.p <= 1 + 1e-9):
            raise ValueError(f"Probability must be in [0, 1], got {self.p}")

    def __repr__(self) -> str:
        text_preview = self.text[:50] + "..." if len(self.text) > 50 else self.text
        return f'Item(p={self.p:.3f}, text="{text_preview}")'


class DiscreteDist:
    """A discrete probability distribution over text candidates.

    Invariants:
        - Probabilities sum to 1.0 +/- 1e-6
        - Items are sorted descending by probability
        - All probabilities are in [0, 1]
    """

    def __init__(self, items: List[Item], trace: Dict[str, Any]):
        self._items = list(items)
        self._trace = dict(trace)
        self._validate()

    def _validate(self):
        if not self._items:
            return
        total = sum(it.p for it in self._items)
        if abs(total - 1.0) > 1e-6:
            raise ValueError(f"Probabilities must sum to 1.0, got {total:.6f}")
        for i in range(len(self._items) - 1):
            if self._items[i].p < self._items[i + 1].p - 1e-9:
                raise ValueError(
                    f"Items must be sorted descending by p: "
                    f"item[{i}].p={self._items[i].p} < item[{i+1}].p={self._items[i+1].p}"
                )

    def __getitem__(self, i: int) -> Item:
        return self._items[i]

    def __len__(self) -> int:
        return len(self._items)

    def __repr__(self) -> str:
        return f"DiscreteDist(k={len(self)}, model={self._trace.get('model', 'unknown')})"

    def __iter__(self):
        return iter(self._items)

    @property
    def items(self) -> List[Item]:
        return self._items

    @property
    def trace(self) -> Dict[str, Any]:
        return self._trace

    def argmax(self) -> Item:
        if not self._items:
            raise ValueError("Cannot argmax on empty distribution")
        return self._items[0]

    def sample(self, seed: Optional[int] = None) -> Item:
        if not self._items:
            raise ValueError("Cannot sample from empty distribution")
        rng = random.Random(seed) if seed is not None else random
        weights = [it.p for it in self._items]
        return rng.choices(self._items, weights=weights, k=1)[0]

    def filter_items(self, pred: Callable[[Item], bool]) -> "DiscreteDist":
        filtered = [it for it in self._items if pred(it)]
        if not filtered:
            raise ValueError("Filter removed all items")
        total = sum(it.p for it in filtered)
        renormed = [
            Item(text=it.text, p=it.p / total, meta={**it.meta, "renormalized": True})
            for it in filtered
        ]
        renormed.sort(key=lambda it: -it.p)
        return DiscreteDist(renormed, self._trace)

    def to_dict(self) -> Dict[str, Any]:
        return {
            "items": [{"text": it.text, "p": it.p, "meta": it.meta} for it in self._items],
            "trace": self._trace,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "DiscreteDist":
        items = [Item(text=it["text"], p=it["p"], meta=it.get("meta", {})) for it in data["items"]]
        return cls(items, data.get("trace", {}))
```

```python
def postprocess_responses(
    parsed_responses: List[Dict[str, Any]],
    min_probability: float = 0.03,
    min_k_survivors: int = 3,
    weight_mode: Literal["elicited", "uniform"] = "elicited",
) -> Tuple[List[Item], Dict[str, Any]]:
    """Postprocess parsed LLM responses into normalized Item list.

    Pipeline: Extract & Repair → Floor Filter → Normalize → Sort → Create Items

    Args:
        parsed_responses: List of dicts with 'response' and 'probability' keys
        min_probability: Floor filter — remove items with p < this value
        min_k_survivors: If filtering removes too many, keep this many regardless
        weight_mode: "elicited" uses LLM-assigned weights; "uniform" = equal weights

    Returns:
        Tuple of (items_list, metadata_dict)
    """
    metadata: Dict[str, Any] = {}

    if not parsed_responses:
        return ([], metadata)

    # Step 1: Extract and repair weights
    candidates = []
    for idx, resp in enumerate(parsed_responses):
        response_text = resp.get("response", resp.get("content", ""))
        raw_p = resp.get("probability", resp.get("confidence", 1.0))

        p_clipped, repairs = repair_weight(raw_p)
        candidates.append({
            "response": response_text,
            "p_raw": raw_p,
            "p_clipped": p_clipped,
            "repairs": repairs,
            "idx_orig": idx,
        })

    # Step 2: Floor filter by min_probability (with relaxation if needed)
    survivors = [c for c in candidates if c["p_clipped"] >= min_probability]
    tau_relaxed = False

    if len(survivors) < min_k_survivors:
        tau_relaxed = True
        candidates.sort(key=lambda c: -c["p_clipped"])
        survivors = candidates[:min_k_survivors]

    metadata["tau_relaxed"] = tau_relaxed
    metadata["num_filtered"] = len(candidates) - len(survivors)

    if not survivors:
        return ([], metadata)

    # Step 3: Normalize weights
    if weight_mode == "uniform":
        weights = [1.0 / len(survivors)] * len(survivors)
    else:
        total_p = sum(c["p_clipped"] for c in survivors)
        if total_p < 1e-12:
            # All zeros — fall back to uniform
            weights = [1.0 / len(survivors)] * len(survivors)
        else:
            weights = [c["p_clipped"] / total_p for c in survivors]

    # Step 4: Sort descending by weight (stable sort preserves original order for ties)
    indexed = list(zip(survivors, weights, range(len(weights))))
    indexed.sort(key=lambda x: (-x[1], x[2]))

    # Step 5: Create Item objects
    items = []
    for cand, weight, _ in indexed:
        meta = {
            "p_raw": cand["p_raw"],
            "p_clipped": cand["p_clipped"],
            "repairs": cand["repairs"],
            "idx_orig": cand["idx_orig"],
        }
        items.append(Item(text=cand["response"], p=weight, meta=meta))

    return (items, metadata)
```

# services/vs-engine/prompt_builder.py
"""VS prompt construction for the Verbalized Sampling engine."""

from config import CONFIDENCE_FRAMINGS

VS_TEMPLATE = """You are asked to provide {k} distinct responses to the following prompt.
Each response should represent a genuinely different approach or perspective.

IMPORTANT: Assign a {framing} value to each response, representing
how likely you believe it is to be the best answer. Distribute your
{framing} as uniformly as possible — aim for each value to be
near {tau} (approximately {approx_uniform}). Avoid concentrating {framing}
on any single response. The values must sum to 1.0.

Respond in this exact JSON format:
{{
  "responses": [
    {{"content": "your response text here", "{framing}": 0.XX}},
    ...
  ]
}}

Do not include any text outside the JSON object.

--- PROMPT ---
{user_prompt}"""


def build_vs_prompt(
    user_prompt: str,
    k: int,
    tau: float,
    confidence_framing: str = "confidence",
    system_prompt: str | None = None,
) -> str:
    """Build the full VS prompt that wraps the user's prompt with VS instructions.

    Args:
        user_prompt: The caller's generation prompt
        k: Number of responses to generate
        tau: Soft prompt ceiling — target probability per response
        confidence_framing: How to frame probabilities (confidence, likelihood, etc.)
        system_prompt: Optional system prompt prepended before VS instructions

    Returns:
        Complete prompt string ready to send to LLM
    """
    framing = CONFIDENCE_FRAMINGS.get(confidence_framing, confidence_framing)
    approx_uniform = f"1/{k} = {1/k:.2f}"

    vs_prompt = VS_TEMPLATE.format(
        k=k,
        framing=framing,
        tau=tau,
        approx_uniform=approx_uniform,
        user_prompt=user_prompt,
    )

    if system_prompt:
        return f"{system_prompt}\n\n{vs_prompt}"
    return vs_prompt
```

# services/vs-engine/selection.py
"""Selection strategies for choosing from a VS distribution."""

from typing import Optional
from distribution import DiscreteDist, Item


def select_from_distribution(
    dist: DiscreteDist,
    strategy: str = "argmax",
    human_selection_index: Optional[int] = None,
) -> Item:
    """Select an item from a distribution using the specified strategy.

    Args:
        dist: The distribution to select from
        strategy: "argmax" (deterministic best), "sample" (weighted random), "human" (by index)
        human_selection_index: Required when strategy is "human"

    Returns:
        Selected Item

    Raises:
        ValueError: For unknown strategy, missing index, or out-of-range index
    """
    if strategy == "argmax":
        return dist.argmax()

    if strategy == "sample":
        return dist.sample()

    if strategy == "human":
        if human_selection_index is None:
            raise ValueError("human_selection_index is required when strategy is 'human'")
        if human_selection_index < 0 or human_selection_index >= len(dist):
            raise ValueError(
                f"human_selection_index {human_selection_index} out of range "
                f"for distribution with {len(dist)} items"
            )
        return dist[human_selection_index]

    raise ValueError(f"Unknown strategy: {strategy}. Use 'argmax', 'sample', or 'human'.")
```

"""LLM connection management for the VS Engine.

Routes requests to Ollama (local), Anthropic (cloud), or OpenAI-compatible
endpoints based on model name patterns.
"""

import json
import logging
import os
import re
from typing import Optional

import httpx

logger = logging.getLogger(__name__)

# Model routing patterns
OLLAMA_PATTERNS = re.compile(r"^(qwen|llama|nomic|mistral|phi|gemma|codellama)")
ANTHROPIC_PATTERNS = re.compile(r"^claude-")
OPENAI_PATTERNS = re.compile(r"^(gpt-|o1-|o3-)")


def detect_provider(model: str) -> str:
    """Detect which provider handles a given model name.

    Returns: "ollama", "anthropic", or "openai"
    """
    if OLLAMA_PATTERNS.match(model):
        return "ollama"
    if ANTHROPIC_PATTERNS.match(model):
        return "anthropic"
    if OPENAI_PATTERNS.match(model):
        return "openai"
    # Default to ollama for unknown models (local-first strategy)
    return "ollama"


async def generate_with_ollama(
    prompt: str,
    model: str,
    ollama_url: str,
    temperature: float = 1.0,
    timeout: float = 120.0,
) -> str:
    """Generate text using Ollama's /api/generate endpoint.

    Args:
        prompt: The full VS-wrapped prompt
        model: Ollama model name (e.g., "qwen3:14b")
        ollama_url: Base URL for Ollama (e.g., "http://dhg-ollama:11434")
        temperature: Sampling temperature
        timeout: Request timeout in seconds

    Returns:
        Raw response text from the LLM
    """
    async with httpx.AsyncClient() as client:
        response = await client.post(
            f"{ollama_url}/api/generate",
            json={
                "model": model,
                "prompt": prompt,
                "stream": False,
                "options": {"temperature": temperature},
            },
            timeout=timeout,
        )
        response.raise_for_status()
        return response.json().get("response", "")


async def generate_with_anthropic(
    prompt: str,
    model: str,
    system_prompt: Optional[str] = None,
    temperature: float = 1.0,
    timeout: float = 120.0,
) -> str:
    """Generate text using the Anthropic API.

    Args:
        prompt: The VS-wrapped prompt (without system prompt — that's passed separately)
        model: Anthropic model name (e.g., "claude-sonnet-4-20250514")
        system_prompt: Optional system prompt
        temperature: Sampling temperature
        timeout: Request timeout in seconds

    Returns:
        Raw response text from the LLM
    """
    try:
        import anthropic
    except ImportError:
        raise RuntimeError("anthropic package not installed. Install with: pip install anthropic")

    api_key = os.getenv("ANTHROPIC_API_KEY")
    if not api_key:
        raise RuntimeError("ANTHROPIC_API_KEY environment variable not set")

    client = anthropic.AsyncAnthropic(api_key=api_key, timeout=timeout)
    messages = [{"role": "user", "content": prompt}]

    response = await client.messages.create(
        model=model,
        max_tokens=4096,
        temperature=temperature,
        system=system_prompt or "",
        messages=messages,
    )
    return response.content[0].text


async def generate(
    prompt: str,
    model: str,
    ollama_url: str,
    system_prompt: Optional[str] = None,
    temperature: float = 1.0,
    timeout: float = 120.0,
) -> str:
    """Route a generation request to the appropriate LLM provider.

    Args:
        prompt: The VS-wrapped prompt
        model: Model name (determines routing)
        ollama_url: Base URL for Ollama
        system_prompt: Optional system prompt (used by Anthropic)
        temperature: Sampling temperature
        timeout: Request timeout in seconds

    Returns:
        Raw response text from the LLM
    """
    provider = detect_provider(model)
    logger.info(f"Routing to {provider} for model {model}")

    if provider == "ollama":
        full_prompt = f"{system_prompt}\n\n{prompt}" if system_prompt else prompt
        return await generate_with_ollama(full_prompt, model, ollama_url, temperature, timeout)

    if provider == "anthropic":
        return await generate_with_anthropic(prompt, model, system_prompt, temperature, timeout)

    # OpenAI-compatible — future implementation
    raise NotImplementedError(f"OpenAI-compatible provider not yet implemented for model: {model}")

# In-memory distribution cache (TTL: 1 hour, max: 1000)
CACHE_TTL_SECONDS = 3600
CACHE_MAX_SIZE = 1000
_distribution_cache: Dict[str, Dict[str, Any]] = {}

SUPPORTED_VARIANTS = {"standard"}


def _cache_distribution(dist_id: str, dist_data: Dict[str, Any]):
    """Store a distribution in the cache with TTL."""
    # Evict expired entries if cache is full
    if len(_distribution_cache) >= CACHE_MAX_SIZE:
        now = time.time()
        expired = [k for k, v in _distribution_cache.items() if now - v["cached_at"] > CACHE_TTL_SECONDS]
        for k in expired:
            del _distribution_cache[k]
        # If still full after evicting expired, remove oldest
        if len(_distribution_cache) >= CACHE_MAX_SIZE:
            oldest_key = min(_distribution_cache, key=lambda k: _distribution_cache[k]["cached_at"])
            del _distribution_cache[oldest_key]

    _distribution_cache[dist_id] = {"data": dist_data, "cached_at": time.time()}
    DISTRIBUTIONS_CACHED.set(len(_distribution_cache))


def _get_cached_distribution(dist_id: str) -> Optional[Dict[str, Any]]:
    """Retrieve a distribution from cache. Returns None if not found or expired."""
    entry = _distribution_cache.get(dist_id)
    if entry is None:
        return None
    if time.time() - entry["cached_at"] > CACHE_TTL_SECONDS:
        del _distribution_cache[dist_id]
        DISTRIBUTIONS_CACHED.set(len(_distribution_cache))
        return None
    return entry["data"]

def _parse_llm_json(raw: str) -> List[Dict[str, Any]]:
    """Parse JSON from LLM response, handling markdown fences and quirks."""
    text = raw.strip()
    # Strip markdown code fences
    if text.startswith("```"):
        text = re.sub(r"^```(?:json)?\s*\n?", "", text)
        text = re.sub(r"\n?```\s*$", "", text)
    parsed = json.loads(text)
    responses = parsed.get("responses", parsed if isinstance(parsed, list) else [])
    result = []
    for r in responses:
        content = r.get("content", r.get("text", r.get("response", "")))
        prob = r.get("confidence", r.get("probability", r.get("likelihood", 0.2)))
        result.append({"response": content, "probability": prob})
    return result


@app.post("/vs/generate", response_model=GenerateResponse)
async def vs_generate(req: GenerateRequest):
    # Validate variant
    if req.variant not in SUPPORTED_VARIANTS:
        raise HTTPException(
            status_code=422,
            detail=f"Unsupported variant '{req.variant}'. Supported: {SUPPORTED_VARIANTS}",
        )

    # Resolve parameters from phase defaults
    defaults = get_phase_defaults(req.phase)
    k = req.k or defaults["k"]
    tau = req.tau if req.tau is not None else defaults["tau"]
    min_probability = req.min_probability if req.min_probability is not None else defaults["min_probability"]
    model = req.model or defaults["model"]
    framing = req.confidence_framing or defaults["confidence_framing"]
    min_k_survivors = req.min_k_survivors or max(k - 2, 2)

    # Build VS prompt
    vs_prompt = build_vs_prompt(
        user_prompt=req.prompt,
        k=k,
        tau=tau,
        confidence_framing=framing,
        system_prompt=req.system_prompt,
    )

    # Generate via LLM
    ollama_url = get_ollama_url()
    start_time = time.time()

    try:
        raw_response = await generate(
            prompt=vs_prompt,
            model=model,
            ollama_url=ollama_url,
            system_prompt=req.system_prompt if model.startswith("claude") else None,
            temperature=req.temperature,
            timeout=120.0,
        )
    except httpx.ConnectError as e:
        logger.error(f"Ollama unreachable: {e}")
        raise HTTPException(status_code=503, detail={"error": "ollama_unavailable", "detail": str(e)})
    except httpx.TimeoutException as e:
        logger.error(f"LLM timeout: {e}")
        raise HTTPException(status_code=503, detail={"error": "llm_timeout", "detail": str(e)})
    except Exception as e:
        logger.error(f"LLM generation failed: {e}")
        raise HTTPException(status_code=502, detail=f"LLM generation failed: {str(e)}")

    duration = time.time() - start_time
    GENERATION_DURATION.observe(duration)

    # Parse LLM response
    try:
        parsed_responses = _parse_llm_json(raw_response)
    except (json.JSONDecodeError, Exception) as e:
        logger.warning(f"JSON parse failed, retrying: {e}")
        # Retry once with a clarification prompt
        try:
            retry_prompt = f"Your previous response was not valid JSON. Please return ONLY a valid JSON object with a 'responses' array. Original prompt:\n\n{vs_prompt}"
            raw_response = await generate(
                prompt=retry_prompt, model=model, ollama_url=ollama_url, temperature=0.5, timeout=120.0,
            )
            parsed_responses = _parse_llm_json(raw_response)
        except Exception as retry_err:
            raise HTTPException(
                status_code=502,
                detail=f"LLM returned unparseable JSON after retry: {str(retry_err)}. Raw: {raw_response[:500]}",
            )

    # Postprocess into DiscreteDist
    items, meta = postprocess_responses(
        parsed_responses, min_probability=min_probability, min_k_survivors=min_k_survivors,
    )

    if not items:
        raise HTTPException(status_code=502, detail="No valid items after postprocessing")

    # Track repairs
    for item in items:
        if item.meta.get("repairs"):
            REPAIR_WEIGHT_TOTAL.labels(model=model).inc()

    # Track tau relaxation and filtering
    if meta.get("tau_relaxed"):
        TAU_RELAXED_TOTAL.labels(phase=req.phase).inc()
    if meta.get("num_filtered", 0) > 0:
        ITEMS_FILTERED_TOTAL.labels(phase=req.phase).inc(meta["num_filtered"])

    # Assign labels (conventional/novel/exploratory) based on probability relative to tau
    for item in items:
        if item.p >= 0.07:
            item.meta["label"] = "conventional"
        elif item.p >= 0.04:
            item.meta["label"] = "novel"
        else:
            item.meta["label"] = "exploratory"

    # Create distribution
    dist_id = str(uuid.uuid4())
    now = datetime.now(timezone.utc).isoformat()

    GENERATIONS_TOTAL.labels(phase=req.phase, model=model).inc()

    # Build response
    response_data = {
        "distribution_id": dist_id,
        "model": model,
        "phase": req.phase,
        "k": k,
        "tau": tau,
        "items": [
            {"content": it.text, "probability": it.p, "metadata": it.meta}
            for it in items
        ],
        "sum_probability": sum(it.p for it in items),
        "tau_relaxed": meta.get("tau_relaxed", False),
        "num_filtered": meta.get("num_filtered", 0),
        "created_at": now,
    }

    # Cache the distribution for /vs/select
    _cache_distribution(dist_id, response_data)

    return response_data


@app.post("/vs/select", response_model=SelectResponse)
async def vs_select(req: SelectRequest):
    cached = _get_cached_distribution(req.distribution_id)
    if cached is None:
        raise HTTPException(
            status_code=404,
            detail="Distribution expired or never existed",
        )

    # Reconstruct items
    items_data = cached["items"]
    items = [
        Item(text=it["content"], p=it["probability"], meta=it.get("metadata", {}))
        for it in items_data
    ]

    if not items:
        raise HTTPException(status_code=404, detail="Distribution has no items")

    # Validate human selection
    if req.strategy == "human" and req.human_selection_index is None:
        raise HTTPException(
            status_code=422,
            detail="human_selection_index is required when strategy is 'human'",
        )
    if req.strategy == "human" and req.human_selection_index is not None:
        if req.human_selection_index < 0 or req.human_selection_index >= len(items):
            raise HTTPException(
                status_code=422,
                detail=f"Index must be >= 0 and < {len(items)}",
            )

    # Create a temporary DiscreteDist for selection
    dist = DiscreteDist(items, trace={"model": cached.get("model", "unknown")})

    try:
        selected = select_from_distribution(
            dist, strategy=req.strategy, human_selection_index=req.human_selection_index,
        )
    except ValueError as e:
        raise HTTPException(status_code=422, detail=str(e))

    SELECTIONS_TOTAL.labels(strategy=req.strategy).inc()

    return {
        "selected": {
            "content": selected.text,
            "probability": selected.p,
            "metadata": selected.meta,
        },
        "strategy_used": req.strategy,
        "distribution_id": req.distribution_id,
    }

# services/vs-engine/evaluators/diversity.py
# Copyright 2025 CHATS-Lab. All Rights Reserved.
# Ported to DHG AI Factory under Apache License 2.0.
# Adapted: OpenAI text-embedding-3-small → Ollama nomic-embed-text
#
# Original: https://github.com/CHATS-lab/verbalized-sampling
# License: Apache 2.0

"""DiversityEvaluator — measures pairwise cosine diversity of VS outputs.

Uses embedding-based pairwise cosine similarity. Adapted from CHATS-lab
analysis/evals/diversity.py to use Ollama nomic-embed-text instead of
OpenAI text-embedding-3-small.
"""

import logging
from typing import Any, Dict, List

import numpy as np

logger = logging.getLogger(__name__)


def compute_pairwise_diversity(embeddings: List[List[float]]) -> Dict[str, float]:
    """Compute pairwise cosine diversity from embedding vectors.

    Args:
        embeddings: List of embedding vectors (each a list of floats)

    Returns:
        Dict with avg_diversity, min_diversity, max_diversity, std_diversity
    """
    if len(embeddings) <= 1:
        return {
            "avg_diversity": 0.0,
            "min_diversity": 0.0,
            "max_diversity": 0.0,
            "std_diversity": 0.0,
        }

    # Convert to numpy and normalize
    emb_array = np.array(embeddings, dtype=np.float64)
    norms = np.linalg.norm(emb_array, axis=1, keepdims=True)
    # Avoid division by zero
    norms = np.maximum(norms, 1e-12)
    normalized = emb_array / norms

    # Compute cosine similarity matrix
    sim_matrix = normalized @ normalized.T
    # Clip to valid range
    sim_matrix = np.clip(sim_matrix, -1.0, 1.0)

    # Extract upper triangle (no diagonal, no double-counting)
    diversities = []
    n = len(embeddings)
    for i in range(n):
        for j in range(i + 1, n):
            similarity = float(sim_matrix[i, j])
            diversity = 1.0 - max(similarity, 0.0)  # diversity = 1 - clip(sim, 0, 1)
            diversities.append(diversity)

    return {
        "avg_diversity": float(np.mean(diversities)),
        "min_diversity": float(np.min(diversities)),
        "max_diversity": float(np.max(diversities)),
        "std_diversity": float(np.std(diversities)),
    }


def compute_text_metrics(texts: List[str]) -> Dict[str, Any]:
    """Compute text-level diversity metrics (vocabulary richness, lengths).

    Args:
        texts: List of response texts

    Returns:
        Dict with avg_response_length, vocabulary_richness, avg_unique_words
    """
    if not texts:
        return {
            "avg_response_length": 0,
            "vocabulary_richness": 0.0,
            "avg_unique_words": 0,
        }

    word_counts = []
    unique_counts = []
    all_words: List[str] = []

    for text in texts:
        words = text.split()
        word_counts.append(len(words))
        unique_counts.append(len(set(words)))
        all_words.extend(words)

    total_words = len(all_words)
    total_unique = len(set(all_words))

    return {
        "avg_response_length": int(np.mean(word_counts)) if word_counts else 0,
        "vocabulary_richness": total_unique / total_words if total_words > 0 else 0.0,
        "avg_unique_words": int(np.mean(unique_counts)) if unique_counts else 0,
    }
```

"""TTCTEvaluator — Torrance Tests of Creative Thinking.

LLM-as-judge evaluation scoring responses on four dimensions:
Fluency, Flexibility, Originality, Elaboration (each 1-5 scale).

Adapted from CHATS-lab analysis/evals/quality.py to use
Claude Sonnet or Ollama as judge instead of GPT-4.1.
"""

import json
import logging
from typing import Any, Dict

logger = logging.getLogger(__name__)

TTCT_WEIGHTS = {
    "fluency": 0.25,
    "flexibility": 0.25,
    "originality": 0.25,
    "elaboration": 0.25,
}


def compute_composite_score(scores: Dict[str, int]) -> float:
    """Compute weighted composite TTCT score.

    Args:
        scores: Dict with fluency, flexibility, originality, elaboration (each 1-5)

    Returns:
        Weighted average (1.0 - 5.0 scale)
    """
    return sum(scores[dim] * TTCT_WEIGHTS[dim] for dim in TTCT_WEIGHTS)


def parse_judge_response(raw: Dict[str, Any]) -> Dict[str, Any]:
    """Parse structured judge response into flat scores with composite.

    Args:
        raw: Dict with fluency/flexibility/originality/elaboration, each having score + justification

    Returns:
        Dict with dimension scores, justifications, and composite

    Raises:
        KeyError: If any required dimension is missing
    """
    result: Dict[str, Any] = {}
    scores: Dict[str, int] = {}

    for dim in TTCT_WEIGHTS:
        dim_data = raw[dim]  # Raises KeyError if missing
        score = dim_data["score"]
        result[dim] = score
        result[f"{dim}_justification"] = dim_data.get("justification", "")
        scores[dim] = score

    result["composite"] = compute_composite_score(scores)
    return result


def build_evaluation_prompt(original_prompt: str, response_text: str) -> str:
    """Build the TTCT evaluation prompt for the judge model.

    Args:
        original_prompt: The prompt that generated the response
        response_text: The response to evaluate

    Returns:
        Complete evaluation prompt string
    """
    return f"""You are an expert evaluator using the Torrance Tests of Creative Thinking (TTCT) framework. Evaluate the following response across four key dimensions of creativity.

REQUEST PROMPT:
{original_prompt}

RESPONSE TO EVALUATE:
{response_text}

EVALUATION RUBRICS:

## 1. FLUENCY
**Definition**: Quality and relevance of the response content that demonstrates productive thinking.
**Scoring (1-5 scale)**:
- 5 (Exceptional): Highly meaningful, directly relevant, clear understanding
- 4 (Strong): Meaningful and relevant with minor gaps
- 3 (Adequate): Some meaningful elements, some off-topic
- 2 (Limited): Few meaningful elements, largely tangential
- 1 (Poor): Fails basic relevance and meaning criteria

## 2. FLEXIBILITY
**Definition**: Breadth of thinking and conceptual diversity demonstrated.
**Scoring (1-5 scale)**:
- 5 (Exceptional): Multiple distinct perspectives and conceptual frameworks
- 4 (Strong): Good variety in approaches with some diversity
- 3 (Adequate): Moderate variety, tends toward similar concepts
- 2 (Limited): Narrow pattern with little variation
- 1 (Poor): Rigid, single-approach thinking

## 3. ORIGINALITY
**Definition**: Statistical rarity and uniqueness compared to typical responses.
**Scoring (1-5 scale)**:
- 5 (Exceptional): Highly unique, novel connections, statistically rare
- 4 (Strong): Several uncommon or surprising elements
- 3 (Adequate): Mix of common and less common ideas
- 2 (Limited): Mostly conventional with occasional less common elements
- 1 (Poor): Highly predictable, common response

## 4. ELABORATION
**Definition**: Degree of detail, development, and descriptive richness.
**Scoring (1-5 scale)**:
- 5 (Exceptional): Rich detail, vivid descriptions, specific examples
- 4 (Strong): Good detail and development with specificity
- 3 (Adequate): Moderate detail, basic ideas somewhat developed
- 2 (Limited): Minimal detail beyond basic concepts
- 1 (Poor): Bare-bones response with no elaboration

INSTRUCTIONS:
1. Read the response carefully
2. For each dimension, provide a score (1-5) with clear justification
3. Be thorough, specific, and evidence-based

Return ONLY a JSON object with this exact structure:
{{
  "fluency": {{"score": N, "justification": "..."}},
  "flexibility": {{"score": N, "justification": "..."}},
  "originality": {{"score": N, "justification": "..."}},
  "elaboration": {{"score": N, "justification": "..."}}
}}"""

from evaluators.diversity import compute_pairwise_diversity, compute_text_metrics
from evaluators.ttct import build_evaluation_prompt, parse_judge_response
from llm_router import embed_with_ollama

# Prometheus metrics for evaluation
DIVERSITY_SCORE = Histogram("vs_diversity_score", "Distribution of avg_diversity scores", ["phase"])
TTCT_COMPOSITE = Histogram("vs_ttct_composite", "Distribution of TTCT composite scores", ["phase"])


class EvaluateItemInput(BaseModel):
    content: str
    probability: float = 0.0


class EvaluateRequest(BaseModel):
    items: List[EvaluateItemInput]
    evaluators: List[str] = ["diversity"]
    judge_model: str = "claude-sonnet-4-20250514"
    embedding_model: str = "nomic-embed-text"
    original_prompt: str = ""
    phase: str = "custom"


@app.post("/vs/evaluate")
async def vs_evaluate(req: EvaluateRequest):
    if len(req.items) < 2:
        raise HTTPException(status_code=422, detail="Evaluation requires at least 2 items")

    result: Dict[str, Any] = {}
    ollama_url = get_ollama_url()

    # Diversity evaluation
    if "diversity" in req.evaluators:
        try:
            embeddings = []
            for item in req.items:
                emb = await embed_with_ollama(item.content, req.embedding_model, ollama_url)
                embeddings.append(emb)

            div_metrics = compute_pairwise_diversity(embeddings)
            text_metrics = compute_text_metrics([it.content for it in req.items])
            result["diversity"] = {**div_metrics, **text_metrics}

            DIVERSITY_SCORE.labels(phase=req.phase).observe(div_metrics["avg_diversity"])
        except Exception as e:
            logger.error(f"Diversity evaluation failed: {e}")
            result["diversity"] = {"error": str(e)}

    # TTCT evaluation
    if "ttct" in req.evaluators:
        try:
            per_item_scores = []
            for idx, item in enumerate(req.items):
                eval_prompt = build_evaluation_prompt(req.original_prompt, item.content)
                raw_judge = await generate(
                    prompt=eval_prompt,
                    model=req.judge_model,
                    ollama_url=ollama_url,
                    temperature=0.1,
                    timeout=120.0,
                )
                # Strip markdown fences if present
                judge_text = raw_judge.strip()
                if judge_text.startswith("```"):
                    judge_text = re.sub(r"^```(?:json)?\s*\n?", "", judge_text)
                    judge_text = re.sub(r"\n?```\s*$", "", judge_text)
                parsed = json.loads(judge_text)
                scores = parse_judge_response(parsed)
                per_item_scores.append({
                    "item_index": idx,
                    "fluency": scores["fluency"],
                    "flexibility": scores["flexibility"],
                    "originality": scores["originality"],
                    "elaboration": scores["elaboration"],
                    "composite": scores["composite"],
                    "justification": "; ".join(
                        f"{dim}: {scores.get(f'{dim}_justification', '')}"
                        for dim in ["fluency", "flexibility", "originality", "elaboration"]
                    ),
                })

            # Aggregate
            if per_item_scores:
                avg_scores = {
                    dim: sum(s[dim] for s in per_item_scores) / len(per_item_scores)
                    for dim in ["fluency", "flexibility", "originality", "elaboration", "composite"]
                }
                result["ttct"] = {**avg_scores, "per_item_scores": per_item_scores}

                TTCT_COMPOSITE.labels(phase=req.phase).observe(avg_scores["composite"])
        except Exception as e:
            logger.error(f"TTCT evaluation failed: {e}")
            # If TTCT was the only evaluator requested, return 503
            if req.evaluators == ["ttct"]:
                raise HTTPException(status_code=503, detail=f"TTCT judge unavailable: {str(e)}")
            result["ttct"] = {"error": str(e)}

    result["evaluated_at"] = datetime.now(timezone.utc).isoformat()
    return result

# langgraph_workflows/dhg-agents-cloud/src/vs_client.py
"""VS Engine client for LangGraph agents.

Provides async functions to call the dhg-vs-engine service for divergent
generation via Verbalized Sampling. Gracefully degrades when the VS engine
is unavailable — agents fall back to standard single-output generation.

Usage in any agent node:
    from vs_client import vs_generate, vs_select, vs_is_available

    result = await vs_generate(prompt="...", phase="gap_analysis", k=5)
    if result:
        selected = await vs_select(result["distribution_id"], strategy="sample")
"""

import os
import logging
from typing import Any, Dict, Optional

import httpx

logger = logging.getLogger(__name__)

VS_ENGINE_URL = os.environ.get("VS_ENGINE_URL", "http://dhg-vs-engine:8000")
VS_TIMEOUT = float(os.environ.get("VS_TIMEOUT", "120.0"))  # Match server-side timeout (spec Section 8.4)


async def vs_is_available() -> bool:
    """Check if the VS engine is reachable."""
    try:
        async with httpx.AsyncClient(timeout=5.0) as client:
            resp = await client.get(f"{VS_ENGINE_URL}/health")
            return resp.status_code == 200
    except Exception:
        return False


async def vs_generate(
    prompt: str,
    phase: str = "custom",
    k: Optional[int] = None,
    tau: Optional[float] = None,
    min_probability: Optional[float] = None,
    model: Optional[str] = None,
    system_prompt: Optional[str] = None,
    confidence_framing: Optional[str] = None,
) -> Optional[Dict[str, Any]]:
    """Call /vs/generate to produce a distribution of diverse outputs.

    Returns None if the VS engine is unavailable (graceful degradation).
    The calling agent should fall back to standard LLM generation.
    """
    body: Dict[str, Any] = {"prompt": prompt, "phase": phase}
    if k is not None:
        body["k"] = k
    if tau is not None:
        body["tau"] = tau
    if min_probability is not None:
        body["min_probability"] = min_probability
    if model is not None:
        body["model"] = model
    if system_prompt is not None:
        body["system_prompt"] = system_prompt
    if confidence_framing is not None:
        body["confidence_framing"] = confidence_framing

    try:
        async with httpx.AsyncClient(timeout=VS_TIMEOUT) as client:
            resp = await client.post(f"{VS_ENGINE_URL}/vs/generate", json=body)
            resp.raise_for_status()
            return resp.json()
    except httpx.ConnectError:
        logger.warning("VS engine unavailable at %s — falling back to standard generation", VS_ENGINE_URL)
        return None
    except httpx.TimeoutException:
        logger.warning("VS engine timed out — falling back to standard generation")
        return None
    except Exception as e:
        logger.error("VS engine error: %s — falling back to standard generation", e)
        return None


async def vs_select(
    distribution_id: str,
    strategy: str = "sample",
    human_selection_index: Optional[int] = None,
) -> Optional[Dict[str, Any]]:
    """Call /vs/select to pick one item from a cached distribution.

    Returns None if the VS engine is unavailable.
    """
    body: Dict[str, Any] = {
        "distribution_id": distribution_id,
        "strategy": strategy,
    }
    if human_selection_index is not None:
        body["human_selection_index"] = human_selection_index

    try:
        async with httpx.AsyncClient(timeout=VS_TIMEOUT) as client:
            resp = await client.post(f"{VS_ENGINE_URL}/vs/select", json=body)
            resp.raise_for_status()
            return resp.json()
    except Exception as e:
        logger.warning("VS select failed: %s", e)
        return None
```

import type { VSDistribution, VSItem } from "./types";

interface VSAlternativesProps {
  distribution: VSDistribution;
  agentLabel: string;
  onSelect?: (item: VSItem, originalIndex: number) => void;
}

// DHG brand colors per spec Section 11.3 and .claude/rules/dhg-brand.md
const LABEL_STYLES: Record<string, string> = {
  conventional: "bg-dhg-purple/10 text-dhg-purple border-dhg-purple/20",
  novel: "bg-dhg-orange/10 text-dhg-orange border-dhg-orange/20",
  exploratory: "bg-dhg-graphite/10 text-dhg-graphite border-dhg-graphite/20",
};

function shuffleArray<T>(arr: T[], seed: number): T[] {
  const shuffled = [...arr];
  let s = seed;
  for (let i = shuffled.length - 1; i > 0; i--) {
    s = (s * 16807 + 0) % 2147483647;
    const j = s % (i + 1);
    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
  }
  return shuffled;
}

export function VSAlternatives({ distribution, agentLabel, onSelect }: VSAlternativesProps) {
  const [expanded, setExpanded] = useState(false);

  if (!distribution || !distribution.items || distribution.items.length <= 1) {
    return null;
  }

  // Shuffle items to eliminate positional bias — use distribution_id as deterministic seed
  // Multiplicative hash (djb2-style) for better entropy than simple char code sum
  const seed = distribution.distribution_id
    .split("")
    .reduce((acc, c) => (acc * 31 + c.charCodeAt(0)) | 0, 0);
  const shuffledItems = shuffleArray(distribution.items, seed);

  return (
    <div className="mt-3 border border-border rounded-lg overflow-hidden">
      <button
        onClick={() => setExpanded(!expanded)}
        className="w-full flex items-center justify-between px-4 py-2 bg-muted/30 hover:bg-muted/50 transition-colors text-sm"
      >
        <span className="flex items-center gap-2 text-muted-foreground">
          <Sparkles className="h-4 w-4 text-dhg-purple" />
          {distribution.items.length} VS alternatives for {agentLabel}
          {distribution.tau_relaxed && (
            <Badge variant="outline" className="text-xs">tau relaxed</Badge>
          )}
        </span>
        {expanded ? <ChevronUp className="h-4 w-4" /> : <ChevronDown className="h-4 w-4" />}
      </button>

      {expanded && (
        <div className="p-3 space-y-2">
          {shuffledItems.map((item, idx) => {
            // The first item in the original (pre-shuffle) distribution is the auto-selected one
            const originalIndex = distribution.items.indexOf(item);
            const isAutoSelected = originalIndex === 0;
            return (
              <VSAlternativeCard
                key={idx}
                item={item}
                isAutoSelected={isAutoSelected}
                onClick={onSelect ? () => onSelect(item, originalIndex) : undefined}
              />
            );
          })}
          <div className="text-xs text-muted-foreground pt-1">
            Model: {distribution.model} · Phase: {distribution.phase} · k={distribution.k} · τ={distribution.tau}
          </div>
        </div>
      )}
    </div>
  );
}

function VSAlternativeCard({ item, isAutoSelected, onClick }: { item: VSItem; isAutoSelected: boolean; onClick?: () => void }) {
  const label = item.metadata?.label ?? "conventional";
  const labelStyle = LABEL_STYLES[label] ?? LABEL_STYLES.conventional;
  const confidencePct = (item.probability * 100).toFixed(0);

  return (
    <Card
      className={`border-border/50 ${isAutoSelected ? "border-l-2 border-l-dhg-purple" : ""} ${onClick ? "cursor-pointer hover:bg-muted/30 transition-colors" : ""}`}
      onClick={onClick}
    >
      <CardContent className="p-3">
        <div className="flex items-start gap-2">
          <div className="flex-1">
            <div className="flex items-center gap-2 mb-1">
              <Badge variant="outline" className={`text-xs ${labelStyle}`}>
                {label}
              </Badge>
              <span className="text-xs text-muted-foreground">
                {confidencePct}% confidence
              </span>
              {isAutoSelected && (
                <span className="text-xs text-dhg-purple font-medium">auto-selected</span>
              )}
            </div>
            <p className="text-sm whitespace-pre-wrap">{item.content}</p>
          </div>
        </div>
      </CardContent>
    </Card>
  );
}
